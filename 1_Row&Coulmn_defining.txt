const int R = 4; // Number of rows
const int C = 4; // Number of columns
int P[R][C];

void init() 
{
    for (int i = 0; i < R; ++i)
        for (int j = 0; j < C; ++j)
            P[i][j] = 0;
}

void update(int r, int c, int s) 
{
    P[r][c] = s;
}

void setup() 
{
    init();
    int row = 1;
    int col = 1;
    int status = 1;
    update(row, col, status);
}
